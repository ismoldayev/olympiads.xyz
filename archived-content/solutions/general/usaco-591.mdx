```java
import java.io.*;
import java.util.*;
public class promote {
	public static void main(String[] args) throws IOException {
		// initialize file I/O
		BufferedReader br = new BufferedReader(new FileReader("promote.in"));
		PrintWriter pw =
		    new PrintWriter(new BufferedWriter(new FileWriter("promote.out")));

		// read in counts for mechanics
		StringTokenizer st = new StringTokenizer(br.readLine());
		int mechanicsBefore = Integer.parseInt(st.nextToken());
		int mechanicsAfter = Integer.parseInt(st.nextToken());

		// read in counts for electromagnetism
		st = new StringTokenizer(br.readLine());
		int electromagnetismBefore = Integer.parseInt(st.nextToken());
		int electromagnetismAfter = Integer.parseInt(st.nextToken());

		// read in counts for thermodynamics
		st = new StringTokenizer(br.readLine());
		int thermodynamicsBefore = Integer.parseInt(st.nextToken());
		int thermodynamicsAfter = Integer.parseInt(st.nextToken());

		// read in counts for modern
		st = new StringTokenizer(br.readLine());
		int modernBefore = Integer.parseInt(st.nextToken());
		int modernAfter = Integer.parseInt(st.nextToken());

		// do the computations
		int thermodynamicsToModern = modernAfter - modernBefore;
		int electromagnetismToThermodynamics =
		    thermodynamicsAfter - thermodynamicsBefore + modernAfter - modernBefore;
		int mechanicsToElectromagnetism = electromagnetismAfter - electromagnetismBefore + thermodynamicsAfter -
		                     thermodynamicsBefore + modernAfter - modernBefore;

		// print the answers
		pw.println(mechanicsToElectromagnetism);
		pw.println(electromagnetismToThermodynamics);
		pw.println(thermodynamicsToModern);

		// close output stream
		pw.close();
	}
}
```
